# /etc/nginx/conf.d/sing.krishposa.com.conf
# Prerequisites:
#   sudo mkdir -p /var/www/certbot/.well-known/acme-challenge
#   sudo certbot certonly --webroot -w /var/www/certbot -d sing.krishposa.com
#   sudo setsebool -P httpd_can_network_connect 1
# OCI: ingress TCP 80 and 443

# --- HTTP: ACME + redirect everything else to HTTPS ---
server {
    listen 80;
    listen [::]:80;
    server_name sing.krishposa.com;

    location ^~ /.well-known/acme-challenge/ {
        root /var/www/certbot;
        default_type "text/plain";
        allow all;
    }

    location / {
        return 301 https://$host$request_uri;
    }

}

# --- HTTPS: karaoke app on 127.0.0.1:8787 ---
server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name sing.krishposa.com;

    ssl_certificate     /etc/letsencrypt/live/sing.krishposa.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/sing.krishposa.com/privkey.pem;

    ssl_session_timeout 1d;
    ssl_session_cache   shared:SSL:10m;
    ssl_protocols       TLSv1.2 TLSv1.3;
    ssl_prefer_server_ciphers off;

    add_header Strict-Transport-Security "max-age=63072000" always;

    # Short URL for host UI (app only serves /karaoke/host.html)
    location = /host {
        return 302 /karaoke/host.html;
    }

    # Stream stems without buffering whole WAV in nginx (smoother phone/laptop playback)
    location /api/out/ {
        proxy_pass http://127.0.0.1:8787;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_buffering off;
        proxy_request_buffering off;
        proxy_read_timeout 3600s;
        proxy_send_timeout 3600s;
    }

    location / {
        proxy_pass http://127.0.0.1:8787;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_read_timeout 3600s;
        proxy_send_timeout 3600s;
        client_max_body_size 200m;
    }
}


sudo nginx -t && sudo systemctl reload nginx
