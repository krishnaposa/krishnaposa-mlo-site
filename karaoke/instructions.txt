  --source karaoke-output \
  --pattern "JOBID/*" \

Manually requeue a job
az storage message put \
  --queue-name karaoke-jobs \
  --content '{"job_id": "JOBID123", "src": {"type":"blob","blob":"JOBID123/original.mp3"}}' \
  --connection-string "$STORAGE_CONN"
