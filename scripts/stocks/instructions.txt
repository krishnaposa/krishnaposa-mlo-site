# scripts/stocks — CLI tools (run from this folder)

## momentum-analyzer.py — momentum portfolio (RS + trailing stop)
python momentum-analyzer.py --seed-from-finviz
python momentum-analyzer.py --seed-from-finviz --merge
python momentum-analyzer.py --tickers-file my_tickers.txt
python momentum-analyzer.py --tickers-file picks.txt --merge
python momentum-analyzer.py

Files: momentum-analyzer.json

## holdings-analyzer.py — holdings list trailing + RS (local JSON)
python holdings-analyzer.py --set-from-file my_tickers.txt
python holdings-analyzer.py --set-from-file picks.txt --merge
python holdings-analyzer.py
python holdings-analyzer.py --remove-on-exit

Files: holdings-list.json, holdings-trailing-state.json

## finviz-screener.py — print / save Finviz screen
python finviz-screener.py
python finviz-screener.py --url "https://finviz.com/screener.ashx?v=111&f=..."
python finviz-screener.py --out picks.txt
python finviz-screener.py --save-json holdings-list.json

## quant-monitor.py — daily quant scores → CSV
python quant-monitor.py
python quant-monitor.py --tickers-file watchlist.txt
python quant-monitor.py --tickers AAPL MSFT
python quant-monitor.py --out-dir ./monitor_out

Output: monitor_out/daily_snapshot_YYYY-MM-DD.csv, leaders_*.csv

## local-list-analyzer.py — edit local watch universe JSON
python local-list-analyzer.py --set-from-file watchlist.txt
python local-list-analyzer.py --show
python local-list-analyzer.py --export watchlist.txt

Files: local-list.json (use with quant-monitor --tickers-file)

## Shared
- Ticker files: same format as my_tickers.txt (txt or csv; see stocks_common.read_symbols_from_file)
- Azure monitor blob: set MONITOR_STORAGE if quant-monitor should use cloud local_list (optional)
