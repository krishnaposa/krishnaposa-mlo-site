sudo bash -c 'cat >/etc/systemd/system/karaoke-agent.service' <<'EOF'
[Unit]
Description=Karaoke Demucs Worker
After=network-online.target

[Service]
Type=simple
User=%i
WorkingDirectory=/home/%i/karaoke-agent
EnvironmentFile=/etc/karaoke-agent.env
ExecStart=/home/%i/venv/bin/python /home/%i/karaoke-agent/agent.py
Restart=always
RestartSec=5

[Install]
WantedBy=multi-user.target
EOF
